# Пример файла .env для Telegram-бота QA Ментор
# 
# ИНСТРУКЦИЯ:
# 1. Скопируйте этот файл и переименуйте в .env (без расширения .txt)
# 2. Замените your_bot_token_here на реальный токен вашего бота
#
# Как получить токен:
# 1. Откройте Telegram
# 2. Найдите бота @BotFather
# 3. Отправьте команду /newbot
# 4. Следуйте инструкциям
# 5. Скопируйте полученный токен
#
# Пример токена выглядит так:
# 1234567890:ABCdefGHIjklMNOpqrsTUVwxyz-1234567890

BOT_TOKEN=your_bot_token_here

# OpenAI API ключ (опционально - для fallback к AI, если ответ не найден в базе знаний)
# Получить можно на https://platform.openai.com/api-keys
# Если не указан, бот будет работать только с локальной базой знаний
OPENAI_API_KEY=sk-your_openai_api_key_here

# Модель OpenAI (опционально, по умолчанию gpt-3.5-turbo)
# OPENAI_MODEL=gpt-3.5-turbo
# или
# OPENAI_MODEL=gpt-4
